name: Deploy Boat Rental Backend

on:
  push:
    branches:
      - main
    paths:
      - 'backend/**'
      - '.github/workflows/deploy.yml'
  pull_request:
    branches:
      - main
    paths:
      - 'backend/**'
  workflow_dispatch:

env:
  NODE_VERSION: '20'
  AWS_REGION: 'us-east-1'

jobs:
  # 1Ô∏è‚É£ Validaci√≥n y Testing
  validate:
    name: Validate & Test
    runs-on: ubuntu-latest
    
    steps:
      - name: üì• Checkout code
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: üîß Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json

      - name: üì¶ Install dependencies
        working-directory: backend
        run: npm ci --prefer-offline --no-audit

      - name: Lint backend code
        working-directory: backend
        run: |
          npm install
          npm run lint

      - name: üèóÔ∏è TypeScript compile check
        working-directory: backend
        run: npm run type-check || echo "No TypeScript configured"

      - name: üß™ Run unit tests
        working-directory: backend
        run: npm run test:unit || echo "No tests configured yet"
        env:
          CI: true

      - name: üìä Upload coverage
        if: always()
        uses: codecov/codecov-action@v3
        with:
          directory: ./backend/coverage
          flags: backend
          fail_ci_if_error: false

  # 2Ô∏è‚É£ Build y An√°lisis de Seguridad
  build-and-scan:
    name: Build & Security Scan
    needs: validate
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: backend
    
    steps:
      - name: üì• Checkout code
        uses: actions/checkout@v4

      - name: üîß Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json

      - name: üì¶ Install dependencies
        run: npm ci --prefer-offline --no-audit

      - name: üèóÔ∏è Build project
        run: npm run build

      - name: üîê Security audit
        run: |
          npm audit --audit-level=high
          npx snyk test --severity-threshold=high || true

      - name: üìè Check bundle size
        run: |
          npm run analyze:size
          # Fail if Lambda size > 50MB
          MAX_SIZE=52428800
          ACTUAL_SIZE=$(du -sb dist | cut -f1)
          if [ $ACTUAL_SIZE -gt $MAX_SIZE ]; then
            echo "‚ùå Bundle size exceeded: $ACTUAL_SIZE bytes"
            exit 1
          fi

      - name: üíæ Cache build artifacts
        uses: actions/cache@v3
        with:
          path: backend/dist
          key: build-${{ github.sha }}

  # 3Ô∏è‚É£ Deploy Preview (PR only)
  deploy-preview:
    name: Deploy Preview Environment
    if: github.event_name == 'pull_request'
    needs: [validate, build-and-scan]
    runs-on: ubuntu-latest
    environment:
      name: preview
      
      url: ${{ steps.deploy.outputs.api_endpoint }}
    defaults:
      run:
        working-directory: backend
    
    steps:
      - name: üì• Checkout code
        uses: actions/checkout@v4

      - name: üîß Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json

      - name: üì¶ Install dependencies
        run: npm ci --production --prefer-offline --no-audit

      - name: üöÄ Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: 
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: üì¶ Install Serverless
        run: npm install -g serverless@3

      - name: üöÄ Deploy to preview
        id: deploy
        run: |
          STAGE="pr-${{ github.event.pull_request.number }}"
          sls deploy --stage $STAGE --verbose
          API_ENDPOINT=$(sls info --stage $STAGE --verbose | grep "endpoint:" | head -1 | awk '{print $2}')
          echo "api_endpoint=$API_ENDPOINT" >> $GITHUB_OUTPUT

      - name: üí¨ Comment PR with preview URL
        uses: actions/github-script@v7
        with:
          script: |
            github.rest.issues.createComment({
              issue_number: context.issue.number,
              owner: context.repo.owner,
              repo: context.repo.repo,
              body: `üöÄ Preview deployed!\n\nAPI Endpoint: ${{ steps.deploy.outputs.api_endpoint }}\n\nThis preview will be automatically destroyed when the PR is closed.`
            })

  # 4Ô∏è‚É£ Deploy Production
  deploy-production:
    name: Deploy to Production
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    needs: [validate, build-and-scan]
    runs-on: ubuntu-latest
    environment:
      name: production
      url: ${{ steps.deploy.outputs.api_endpoint }}
    defaults:
      run:
        working-directory: backend
    
    steps:
      - name: üì• Checkout code
        uses: actions/checkout@v4

      - name: üîß Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'
          cache-dependency-path: backend/package-lock.json

      - name: üì¶ Install production dependencies
        run: npm ci --production --prefer-offline --no-audit

      - name: üíæ Restore build artifacts
        uses: actions/cache@v3
        with:
          path: backend/dist
          key: build-${{ github.sha }}

      - name: üöÄ Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: üì¶ Install Serverless
        run: npm install -g serverless@3

      - name: üè∑Ô∏è Create deployment tag
        run: |
          DEPLOY_TAG="deploy-$(date +%Y%m%d-%H%M%S)-${GITHUB_SHA::8}"
          echo "DEPLOY_TAG=$DEPLOY_TAG" >> $GITHUB_ENV

      - name: üöÄ Deploy to production
        id: deploy
        run: |
          sls deploy --stage prod --verbose
          API_ENDPOINT=$(sls info --stage prod --verbose | grep "endpoint:" | head -1 | awk '{print $2}')
          echo "api_endpoint=$API_ENDPOINT" >> $GITHUB_OUTPUT

      - name: Health check
        run: |
          echo "Waiting for deployment to stabilize..."
          sleep 30
          HEALTH_ENDPOINT="https://aqn47vn9ce.execute-api.us-east-1.amazonaws.com/dev/hello"
          HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" $HEALTH_ENDPOINT)

          if [ $HTTP_CODE -eq 200 ]; then
            echo "‚úÖ Health check passed!"
          else
            echo "‚ùå Health check failed with status $HTTP_CODE"
            exit 1
          fi


      - name: üìä Report metrics to CloudWatch
        run: |
          aws cloudwatch put-metric-data \
            --namespace "BoatRental/Deployments" \
            --metric-name "SuccessfulDeployments" \
            --value 1 \
            --dimensions Environment=production,Service=backend

      - name: üè∑Ô∏è Tag deployment in Git
        run: |
          git config user.name github-actions
          git config user.email github-actions@github.com
          git tag -a "${{ env.DEPLOY_TAG }}" -m "Production deployment"
          git push origin "${{ env.DEPLOY_TAG }}"


  # 5Ô∏è‚É£ Cleanup Preview Environments
  cleanup-preview:
    name: Cleanup Preview Environment
    if: github.event_name == 'pull_request' && github.event.action == 'closed'
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: backend
    
    steps:
      - name: üì• Checkout code
        uses: actions/checkout@v4

      - name: üîß Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}

      - name: üöÄ Configure AWS credentials
        uses: aws-actions/configure-aws-credentials@v4
        with:
          aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
          aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
          aws-region: ${{ env.AWS_REGION }}

      - name: üì¶ Install Serverless
        run: npm install -g serverless@3

      - name: üóëÔ∏è Remove preview environment
        run: |
          STAGE="pr-${{ github.event.pull_request.number }}"
          sls remove --stage $STAGE --verbose || true
